SELECT rm.role_principal_id as [role_id], sp.name as [role_name], sp.type_desc as [role_type], sp.is_disabled as [role_disabled], rm.member_principal_id as [member_id], sp2.name as [member_name], sp2.type_desc as [member_type], sp2.is_disabled as [member_disabled] from sys.server_role_members rm left outer join sys.server_principals sp on rm.role_principal_id = sp.principal_id left outer join sys.server_principals sp2 on rm.member_principal_id = sp2.principal_id ;
